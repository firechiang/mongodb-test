#### Update（更新数据）命令简单使用（注意：先使用命令行连接好 Mongodb 且选择好数据库）
```bash
# 查询修改整条数据（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5的数据 
    {
         _id: "5"
    },
    // 将ID等于5的数据 修改从一下内容
    {
        _id:"5",
        name:"dgfdggdX",
        hobby: ["dasd","sadsa"]
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })
    
# 查询修改某个属性的值（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5和数组hobby里面包含asa的数据
    {
         _id: "5",
         hobby: "asa"
    },
    // 修改属性的值（注意：也可以添加新的属性，直接写到下面即可）
    {
        $set: {
            name: "wenjuan",
            age: 25,
            // 修改内嵌数据的写法（这是修改 desc 对象里面的name属性的值）
            "desc.name": "sdsadasdasd",
            "desc.name1": "法大师傅大师傅",
            // 修改数组属性里面的元素，这个也可以添加元素只要写好下标即可（在这里是将hobby数组的第0个位置的值修改为"a"）
            "hobby.0": "a",
            // 将上面查询匹配（hobby: "asa"）到的元素修改成asas
            "hobby.$": "asas",
            // 修改内嵌数组里的所有元素修改成asas
            //"hobby.0.$[]": "asas"
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })
    
# 查询删除属性（注意：主键是不能删除的）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 删除属性
    {
        $unset: {
            name: "",
            // 删除内嵌数据的写法（这是将 desc 对象里面的name属性删除）
            "desc.name": "",
            // 删除数组属性里面的元素，删除元素只是将元素的值制空而已（在这里是将hobby数组的第0个位置的元素删除）
            "hobby.0": ""
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
# 查询修改属性的名称（注意：主键名是不能修改的）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 修改属性名称
    {
        $rename: {
            name: "name1",
            // 修改内嵌数据的属性名称（这是将 desc 对象里面的name属性名修改为name1）
            "desc.name": "desc.name1",
            // 修改内嵌数据属性的位置（这是将 desc 对象里面的name属性调整到父对象里面去，它的名字是nameD）
            "desc.name": "nameD",
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    }) 
    
    
# 查询修改属性的值，加减法（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 修改数据
    {
        $inc: {
            // 加 1
            age: 1,
            // 减 2
            //age: -2
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
# 查询修改属性的值，乘除法（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 修改数据
    {
        $mul: {
            // 乘2 
            age: 2,
            // 除 2
            //age: 0.5
       }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
# 查询修改属性的值，谁大使用谁，只要是可以比较大小的都可以改（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 修改数据
    {
        $max: {
            // 2和原有的数据比较， 谁大使用谁
            age: 2,
       }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
# 查询修改属性的值，谁小使用谁，只要是可以比较大小的都可以改（注意：主键是不能改的，可以不写）
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    // 修改数据
    {
        $min: {
            // 2和原有的数据比较， 谁小使用谁
            age: 2,
       }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    }) 
    
# 查询添加数组属性的元素
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    {
        // 添加数组元素
        $push: {
            // 添加元素
            hobby: "2"
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })      
    
    
# 查询添加数组属性非重复的元素
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    {
        // 添加数组元素（注意：以下方式不会添加重复元素）
        $addToSet: {
            // 添加元素
            hobby: "2",
            // 将数组里面的元素一个一个的插入到数组，而不是整个数组当成一个元素插入到数组 
            //hobby: {$each: ["adas","asdsadas"]}
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
# 查询删除数组属性的元素，只能删除第一个或最后一个元素
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    {
        $pop: {
            // 删除元素（1 表示删除最后一个元素，-1表示第一个）
            hobby: 1,
            // 数组内嵌数组元素删除（删除第3个内嵌数组的最后一个元素）
            //"hobbySS.3": 1
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    }) 
    
# 查询删除数组属性的元素，删除特定元素
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    {
        $pull: {
            hobby: {
                // 删除数组元素包含a的第一个元素
                $regex: /a/,
                // 数组内嵌数组的删除元素的方法
                //$elemMatch: {$eq: "a"}
           }
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })  
    
    
# 查询删除数组属性的元素，删除内嵌数据，pullAll需完全匹配，pull只需要值相同即可，顺序可以不一样
$ db.test_user.update(
    // 先查询到ID等于5的数据
    {
         _id: "5"
    },
    {
        $pullAll: {
            // 删除内嵌数组它的数组是 ["a","s"]
            hobby: ["a","s"]
        }
    },
    {
        // 是否更新所有匹配的数据 （默认是false，只更新第一条数据，如果是true的话不保证原子性）
        multi: true
    })                              
```
